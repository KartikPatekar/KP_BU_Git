
\subsection*{Part 1}

\begin{align*}
    u = 0.05 \implies f(\theta) = s\frac{1-u}{u} = 0.38\\
\end{align*}

\begin{align*}
    f(\theta) &= M(1, \theta) = \frac{\theta}{(1+\theta^\iota)^{1/\iota}}\\
    \implies \theta = 0.4414
\end{align*}

Hence $\theta = 0.4414$ is consistent with $u = 0.05$.\\

From the free entry and wage condition in steady state
\begin{align*}
    c &= \beta q(\theta) \frac{1}{1 - \beta(1 - s)} [(1 - \gamma)(z - b) - \gamma \theta c]\\
    \implies c &= 1.211
\end{align*}


\subsection*{Part 2}
Transition matrix is given by

\begin{equation*}
    P = \begin{bmatrix}
        \pi~, 1-\pi \\
        1- \pi~~~, \pi
    \end{bmatrix}
\end{equation*}

It has a unique stationary state 
\begin{align*}
    p = \begin{bmatrix}
        0.5 \\ 
        0.5
    \end{bmatrix}
\end{align*}

Since $\mathbb E[\log z] = 0$ and $Var[\log z] = 0.0001$, we have 
\begin{align*}
    \log z_h &= -\log z_l\\
    (\log z_h)^2 + (\log z_l)^2 &= 2*0.0001
\end{align*}

which gives 
$$ z_h = e^{0.01}~~~,~~~ z_l = e^{-0.01}$$

The autocorrelation is 
\begin{align*}
    \rho &= \frac{\mathbb{E}[\log z_t \log z_{t+1}]}{\operatorname{Var}(\log z_t)} \\
    &= \frac{\mathbb{E}\left[ \log z_t \mathbb{E}[\log z_{t+1} \mid \log z_t] \right]}{\operatorname{Var}(\log z_t)}
\end{align*}

Substituting $\mathbb{E}[\log z_{t+1} \mid \log z_t] = \pi \log z_t - (1-\pi)\log z_t$ gives
\begin{align*}
    \rho = 2\pi - 1 \implies \pi = 0.985
\end{align*}

\subsection*{Part 3}

Free entry condition and value of filling a position is 
\begin{align*}
c &= \beta q(\theta(z,u)) \mathbb{E}_{z',u'} [J(z',u')|z] \\
J(z,u) &= z - w(z,u) + \beta(1-s) \mathbb{E}_{z',u'} [J(z',u')|z] \\
\end{align*}
where $w(z,u) = (1-\gamma)b + \gamma(z + \theta(z,u)c)$.\\

Substituting the formul for $J(z,u)$ and $w(z,u)$ in first equation gives two relationships between $c$, $z_h$, $z_l$, $\theta(z_h, u)$ and $\theta(z_l, u)$. For fixed $c$, since these equations do not directly depend on $u$, $\theta$ is a function of $z$ only. Hence, $w$ also does not directly depend on $u$, and $J(z,u)$ is a function of $z$ only. We can write $\theta(z)$ and $J(z)$.

We have
\begin{align*}
    q(\theta) = M(1/\theta, 1) =  \frac{1}{(1+\theta^\iota)^{1/\iota}}   
\end{align*}

The 4 equations in terms of $\{\theta(z_l ), \theta(z_h), J(z_l ), J(z_h)\}$ are

\begin{align*}
    \frac{c}{\beta } (1+\theta(z_h)^\iota)^\frac{1}{\iota} &= \pi J(z_h) + (1-\pi) J(z_l) \\
    \frac{c}{\beta } (1+\theta(z_l)^\iota)^\frac{1}{\iota} &= \pi J(z_l) + (1-\pi) J(z_h) \\
    J(z_h) &= (1-\gamma)(z_h - b) - \gamma c \theta(z_h) + \beta (1-s)[\pi J(z_h) + (1-\pi) J(z_l)]\\
    J(z_l) &= (1-\gamma)(z_l - b) - \gamma c \theta(z_l) + \beta (1-s)[\pi J(z_l) + (1-\pi) J(z_h)]\\    
\end{align*}

Solving these gives
\begin{align*}
    \theta(z_h) &=  0.448925 \\
    \theta(z_l) &= 0.433358 \\
    J(z_h) & = 1.416369 \\
    J(z_l) & = 1.405458 \\
\end{align*}


The simulation results are in figure \ref{fig:q3_sim_b0.4}. The standard deviations are
\begin{align*}
    sd(\log z_t) &= 0.010\\
    sd(\log u_t) &= 0.013\\
    sd(\log v_t) &= 0.007\\
    sd(\log \theta_t) &= 0.018\\    
\end{align*}

Assuming perfect correlation, $|d\log u| \approx 1.3 |d\log u|$, which is much smaller than the value overserved in data.

\subsection*{Part 5}
Let flow utility be $b= \bar b z_t$. Then the 4 equations are 

\begin{align*}
    \frac{c}{\beta } (1+\theta(z_h)^\iota)^\frac{1}{\iota} &= \pi J(z_h) + (1-\pi) J(z_l) \\
    \frac{c}{\beta } (1+\theta(z_l)^\iota)^\frac{1}{\iota} &= \pi J(z_l) + (1-\pi) J(z_h) \\
    J(z_h) &= (1-\gamma)(1 - \bar b)z_h - \gamma c \theta(z_h) + \beta (1-s)[\pi J(z_h) + (1-\pi) J(z_l)]\\
    J(z_l) &= (1-\gamma)(1 - \bar b)z_l - \gamma c \theta(z_l) + \beta (1-s)[\pi J(z_l) + (1-\pi) J(z_h)]\\    
\end{align*}

which gives the solutions
\begin{align*}
    c &=  0.101\\
    \theta(z_h) &= 0.445394 \\
    \theta(z_l) &= 0.436061 \\
\end{align*}


the simulation results are in figure \ref{fig:q3_sim_part5}. The standard deviations are 
\begin{align*}
    sd(\log z_t) &= 0.010\\
    sd(\log u_t) &= 0.008\\
    sd(\log v_t) &= 0.004\\
    sd(\log \theta_t) &= 0.011\\    
\end{align*}

Now we only have $|d\log u| \approx 0.8 |d\log u|$, which worsens the Shimer puzzle. If the outside option is worse during a downturn, then worker becomes more likely to work for lower wage which counters the unemployment effect of low productivity.

\subsection*{Part 6}
Let rigid wages be $w_t = \bar w = 0.95$. Then the 4 equations are 

\begin{align*}
    \frac{c}{\beta } (1+\theta(z_h)^\iota)^\frac{1}{\iota} &= \pi J(z_h) + (1-\pi) J(z_l) \\
    \frac{c}{\beta } (1+\theta(z_l)^\iota)^\frac{1}{\iota} &= \pi J(z_l) + (1-\pi) J(z_h) \\
    J(z_h) &= z_h - \bar w + \beta (1-s)[\pi J(z_h) + (1-\pi) J(z_l)]\\
    J(z_l) &= z_l - \bar w + \beta (1-s)[\pi J(z_l) + (1-\pi) J(z_h)]\\    
\end{align*}

Also, fixed wages, formula for $c$ in steady state changes and can be computed by equations (3-4) and (3-5)
\begin{align*}
    J(z) &= \frac{z-\bar w}{1-\beta(1-s)}\\
    c &= \beta q(\theta)\times \frac{z-\bar w}{1-\beta(1-s)}
\end{align*}
Using $\theta = 0.4414$ as before gives $c= 1.887$ 

which gives the solutions
\begin{align*}
    c &=  1.840\\
    \theta(z_h) &= 0.607849 \\
    \theta(z_l) &= 0.239225 \\
\end{align*}


the simulation results are in figure \ref{fig:q3_sim_part6}. The standard deviations are 
\begin{align*}
    sd(\log z_t) &= 0.010\\
    sd(\log u_t) &= 0.344\\
    sd(\log v_t) &= 0.180\\
    sd(\log \theta_t) &= 0.466\\    
\end{align*}

This shows that completely rigid wages solve the Shimer puzzle, bringing $|d\log u| \approx 0.8 |d\log u|$. With rigid wages, firm's value function of a filled position fluctuates a lot with worker productivity, which results in more firms closing down during recession and more vacancies being created in boom.
